mutation ($id: String!, $body: String!) {
	updateFeedDocument (id: $id, body: $body) {
		__typename
		...on FeedDocument {
			id
			feed
			author
			title
			posted
			body
			body_html
			created
			updated
			url
		}
		...on FeedDocumentDoesNotExistError {
			message
		}
		...on InsufficientPerms {
			message
		}
	}
}
