mutation ($userdata: WeatherUserInput!) {
	createWeatherUser (userdata: $userdata) {
		__typename
		...on UserData {
			username
			display_name
			theme {
				colors {
					name
					value
				}
				sizes {
					name
					value
				}
			}
			perms
			last_login
			groups
			disabled_modules
			email
		}
		...on BadUserNameError {
			message
		}
		...on UserExistsError {
			message
		}
		...on InsufficientPerms {
			message
		}
	}
}
