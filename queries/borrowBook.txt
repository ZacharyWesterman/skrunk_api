mutation ($id: String!) {
	borrowBook (id: $id) {
		__typename
		...on Book {
			id
			title
			subtitle
			authors
			publisher
			publishedDate
			description
			industryIdentifiers {
				type
				identifier
			}
			pageCount
			categories
			maturityRating
			language
			thumbnail
			smallThumbnail
			creator
			owner {
				username
				display_name
			}
			shared
			shareHistory {
				user_id
				name
				display_name
				start
				stop
			}
			rfid
			ownerHistory {
				user_id
				name
				display_name
				start
				stop
			}
			ebooks {
				url
				fileType
			}
			audiobook
		}
		...on BookTagDoesNotExistError {
			message
		}
		...on BookCannotBeShared {
			message
		}
		...on UserDoesNotExistError {
			message
		}
		...on InsufficientPerms {
			message
		}
	}
}
