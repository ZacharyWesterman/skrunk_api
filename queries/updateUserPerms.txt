mutation ($username: String!, $perms: [String!]!) {
	updateUserPerms (username: $username, perms: $perms) {
		__typename
		...on UserData {
			username
			display_name
			theme {
				colors {
					name
					value
				}
				sizes {
					name
					value
				}
			}
			perms
			last_login
			groups
			disabled_modules
		}
		...on UserDoesNotExistError {
			message
		}
		...on InsufficientPerms {
			message
		}
	}
}
