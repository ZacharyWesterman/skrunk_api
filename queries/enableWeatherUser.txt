mutation ($username: String!) {
	enableWeatherUser (username: $username) {
		__typename
		...on UserData {
			username
			display_name
			theme {
				colors {
					name
					value
				}
				sizes {
					name
					value
				}
			}
			perms
			last_login
			groups
		}
		...on UserDoesNotExistError {
			message
		}
		...on InsufficientPerms {
			message
		}
	}
}
