mutation ($username: String!, $display_name: String!) {
	updateUserDisplayName (username: $username, display_name: $display_name) {
		__typename
		...on UserData {
			username
			display_name
			theme {
				colors {
					name
					value
				}
				sizes {
					name
					value
				}
			}
			perms
			last_login
			groups
			disabled_modules
		}
		...on UserDoesNotExistError {
			message
		}
		...on InsufficientPerms {
			message
		}
	}
}
